<!-- Inicio de código para tiempo de cuarentena -->
<script type="text/javascript">
function cerramiento() {
//Asignar el valor de las unidades en milisegundos
var msporminuto = 1000 * 60;
var msporhora = msporminuto * 60;
var mspordia = msporhora * 24;
//Asignar las fechas en milisegundos
var inicio = new Date(2020,2,20);//Fecha de inicio de la cuarentena: 20-3-2020 a la medianoche
var actual = new Date();//Fecha actual
var final = new Date(2020,10,9);//Fecha de finalización de la cuarentena (8-11-2020), tomada a las 0 horas del día posterior
if(final < actual) {actual = final;}
//Obtener la diferencia en milisegundos
var intervalo = actual.getTime() - inicio.getTime();//Da el tiempo transcurrido: avanza el contador
//var intervalo = final.getTime() - actual.getTime();//Da el tiempo faltante: retrocede el contador
//Calcular los días y los milisegundos sobrantes
var dias = Math.floor(intervalo / mspordia );
intervalo = intervalo - (dias * mspordia );
//Calcular las horas y los milisegundos sobrantes
var horas = Math.floor(intervalo / msporhora );
intervalo = intervalo - (horas * msporhora );
//Calcular los minutos y los milisegundos sobrantes
var minutos = Math.floor(intervalo / msporminuto );
intervalo = intervalo - (minutos * msporminuto );
//Calcular los segundos y fijar los datos de horas, minutos y segundos en dos dígitos cada uno
var segundos = Math.floor(intervalo / 1000 );
if(horas < 10) { horas = '0' + horas; }
if(minutos < 10) { minutos = '0' + minutos; }
if(segundos < 10) { segundos = '0' + segundos; }
//Mostrar el resultado
if(final > actual) {var micuarentena = "Tiempo de cuarentena = " + dias + " d " + horas + " h " + minutos + " min " + segundos + " s " + "<br />Récord Guinness 2020 en curso";}
else {var micuarentena = "Tiempo de cuarentena = " + dias + " d " + horas + " h " + minutos + " min " + segundos + " s " + "<br />Récord Guinness 2020 asegurado" + "<br />¡¡¡Fin de la cuarentena!!!";}
//if(final > actual) {var micuarentena = "Tiempo restante = " + dias + " d " + horas + " h " + minutos + " min " + segundos + " s " + "<br />Récord Guinness 2020 en curso";}
//else {var micuarentena = "Tiempo restante = " + dias + " d " + horas + " h " + minutos + " min " + segundos + " s " + "<br />Récord Guinness 2020 asegurado" + "<br />¡¡¡Fin de la cuarentena!!!";}
return micuarentena; }
function actualizar() { //Función del temporizador
var miestado=cerramiento(); //Recoger cantidad de días de encierro
micuarentena=document.getElementById("contenedor"); //Buscar elemento contenedor
micuarentena.innerHTML=miestado; } //Incluir datos de encierro
setInterval(actualizar,1000); //Iniciar temporizador
</script>
<div id="contenedor" class="contenedor"></div>
